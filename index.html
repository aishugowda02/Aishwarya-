int sensorPin = 2; // IR Sensor connected to Pin 2 int motorPin = 9; // Fan Motor connected to Pin 9 void setup() { pinMode(sensorPin, INPUT); pinMode(motorPin, OUTPUT); } void loop() { int val = digitalRead(sensorPin); if (val == LOW) { // If car is detected digitalWrite(motorPin, HIGH); // Turn fan ON } else { digitalWrite(motorPin, LOW); // Turn fan OFF } }
